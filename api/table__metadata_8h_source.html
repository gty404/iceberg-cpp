182 using SnapshotsMapRef = std::reference_wrapper<const SnapshotsMap>;

186 Result<SortOrdersMapRef> GetSortOrdersById() const;
187 Result<SnapshotsMapRef> GetSnapshotsById() const;
188
189 private:
190 static Result<SchemasMap> InitSchemasMap(const TableMetadata* metadata);
191 static Result<PartitionSpecsMap> InitPartitionSpecsMap(const TableMetadata* metadata);
192 static Result<SortOrdersMap> InitSortOrdersMap(const TableMetadata* metadata);
193 static Result<SnapshotsMap> InitSnapshotMap(const TableMetadata* metadata);
194
195 const TableMetadata* metadata_;
196 Lazy<InitSchemasMap> schemas_map_;
197 Lazy<InitPartitionSpecsMap> partition_specs_map_;
198 Lazy<InitSortOrdersMap> sort_orders_map_;
199 Lazy<InitSnapshotMap> snapshot_map_;
200};
